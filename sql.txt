CREATE TABLE IF NOT EXISTS places (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  name          TEXT    NOT NULL,
  category      TEXT    NOT NULL,
  city          TEXT    NOT NULL,
  description   TEXT,
  is_approved   INTEGER NOT NULL DEFAULT 0,   -- 0/1 (SQLite bool)
  created_by    INTEGER,                      -- user_auth.id (если есть)
  created_at    TEXT    NOT NULL DEFAULT (datetime('now')),
  updated_at    TEXT    NOT NULL DEFAULT (datetime('now')),
  avg_rating    REAL    NOT NULL DEFAULT 0.0,
  reviews_count INTEGER NOT NULL DEFAULT 0,
  FOREIGN KEY (created_by) REFERENCES users_auth(id) ON DELETE SET NULL
);

CREATE INDEX IF NOT EXISTS ix_places_city     ON places(city);
CREATE INDEX IF NOT EXISTS ix_places_category ON places(category);
CREATE INDEX IF NOT EXISTS ix_places_approved ON places(is_approved);
